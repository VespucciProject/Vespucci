165 
166

168  plot_button_->setDisabled(false);

172  }

209 void DataViewer::on_exportToolButton_clicked()
210 {
211  QString filename = QFileDialog::getSaveFileName(this,
212  tr("Export As..."),
213  directory_,
214  tr("Comma-separated Variables (*.csv);;"
215  "Tab-delimited Text (*.txt);;"
216  "Armadillo Binary (*.arma);;"));
217  QFileInfo file_info(filename);
218  bool success = false;
219  if (filename.isEmpty())
220  return;
221  workspace->set_directory(file_info.dir().path());
222  QString extension = file_info.suffix();
223  if (extension == "arma")
224  success = current_data_->save(filename.toStdString(), arma_binary);
225  else if (extension == "csv")
226  success = current_data_->save(filename.toStdString(), csv_ascii);
227  else
228  success = current_data_->save(filename.toStdString(), raw_ascii);
229 
230  if (success)
231  QMessageBox::information(this, "Success!", "File " + filename + " written successfully!");
232  else
233  QMessageBox::warning(this, "File Not Saved", "File was not written successfully.");
234 
235  return;
236 }
