41 bool Vespucci::Math::DimensionReduction::plsregress(arma::mat X, arma::mat Y,
42  int components, arma::mat &X_loadings,
43  arma::mat &Y_loadings,
44  arma::mat &X_scores,
45  arma::mat &Y_scores,
46  arma::mat &coefficients,
47  arma::mat &percent_variance,
48  arma::mat &fitted)
49 {
50  using namespace arma;
51  uword observations = X.n_rows;
52  uword predictors = X.n_cols;
53  uword responses = Y.n_cols;
54  //Mean centering
55  mat Xmeans = arma::mean(X);
56  mat Ymeans = arma::mean(Y);
57  //Octave does below with bsxfun. After optimization, this should hopefully not
58  // be slower.
59  X.each_row() -= Xmeans; //element-wise subtraction of mean
60  Y.each_row() -= Ymeans; //same
61 
62  mat S = trans(X) * Y;
63  mat R = zeros(predictors, components);
64  mat P = R;
65  mat V = R;
66  mat T = zeros(observations, components);
67  mat U = T;
68  mat Q = zeros(responses, components);
69  mat eigvec;
70  vec eigval;
71 
72  mat q;
73  mat r;
74  mat t;
75  mat nt;
76  mat p;
77  mat u;
78  mat v;
79  double max_eigval;
80  for (int i = 0; i < components; ++i){
81  eig_sym(eigval, eigvec, (trans(S) * S));
82  max_eigval = eigval.max();
83  uvec dom_index = find(eigval >= max_eigval);
84  uword dominant_index = dom_index(0);
85 
86  q = eigvec.col(dominant_index);
87 
88  r = S*q; //X block factor weights
89  t = X*r; //X block factor weights
90  t.each_row() -= mean(t); //center t
91  nt = arma::sqrt(trans(t)*t); //compute norm (is arma::norm() the same?)
92  t.each_row() /= nt;
93  r.each_row() /= nt; //normalize
94 
95  p = trans(X)*t; //X block factor loadings
96  q = trans(Y)*t; //Y block factor loadings
97  u = Y*q; //Y block factor scores
98  v = p;
99 
100  //Ensure orthogonality
101  if (i > 0){
102  v = v - V*(trans(V)*p);
103  u = u - T*(trans(T)*u);
104  }
105  v.each_row() /= arma::sqrt(trans(v) * v); //normalize orthogonal loadings
106  S = S - v * (trans(v)*S); //deflate S wrt loadings
107  R.col(i) = r;
108  T.col(i) = t;
109  P.col(i) = p;
110  Q.col(i) = q;
111  U.col(i) = u;
112  V.col(i) = v;
113  }
114 
115  //Regression coefficients
116  mat B = R*trans(Q);
117  fitted = T*trans(Q);
118  fitted.each_row() += Ymeans;
119 
120  //Octave creates copies from inputs before sending to output. Doing same
121  //here just to be safe.
122  coefficients = B;
123  X_scores = T;
124  X_loadings = P;
125  Y_scores = U;
126  Y_loadings = Q;
127  //projection = R;
128  percent_variance.set_size(2, coefficients.n_cols);
129  percent_variance.row(0) = sum(arma::abs(P)%arma::abs(P)) / sum(sum(arma::abs(X)%arma::abs(X)));
130  percent_variance.row(1) = sum(arma::abs(Q)%arma::abs(Q)) / sum(sum(arma::abs(Y)%arma::abs(Y)));
131  return true;
132 
133 }
