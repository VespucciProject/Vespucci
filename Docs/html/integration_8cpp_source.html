31 double Vespucci::Math::Quantification::IntegratePeak(const arma::vec &X, arma::uword min_index, arma::uword max_index, double abscissa_step, arma::vec &baseline)
32 {
33  arma::vec region = X.subvec(min_index, max_index);
34 
35  double start = X(min_index);
36  double end = X(max_index);
37 
38  baseline = arma::linspace(start, end, region.n_elem);
39 
40  double baseline_area = sum(baseline) / abscissa_step;
41  double region_area = sum(region) / abscissa_step;
42 
43  return region_area - baseline_area;
44 }

56 arma::vec Vespucci::Math::Quantification::IntegratePeakMat(const arma::mat &X, arma::vec abscissa, double &min, double &max, arma::mat &baselines, arma::uvec &boundaries)
57 {
58  double delta = std::abs(abscissa(1) - abscissa(0));
59  arma::uvec left_bound = find(((min-delta) <= abscissa) && (abscissa <= (min+delta)));
60  arma::uvec right_bound = find(((max-delta) <= abscissa) && (abscissa <= (max+delta)));
61 
62  arma::uword min_index = left_bound(0);
63  arma::uword max_index = right_bound(0);
64  min = abscissa(min_index);
65  max = abscissa(max_index);
66  boundaries << min_index << arma::endr << max_index;
67  arma::vec results(X.n_cols);
68  baselines.set_size(X.col(0).subvec(min_index, max_index).n_elem, X.n_cols);
69  arma::vec baseline(baselines.n_cols);
70  for (arma::uword i = 0; i < X.n_cols; ++i){
71  results(i) = IntegratePeak(X.col(i), min_index, max_index, delta, baseline);
72  baselines.col(i) = baseline;
73  }
74 
75  return results;
76 }
