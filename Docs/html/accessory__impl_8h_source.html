46 template <typename T> T Vespucci::Math::diff(const T &X, arma::uword deriv_order)
47 {
48  T return_value;
49  if (deriv_order == 0){
50  return_value = X;
51  }
52 
53  else if (deriv_order > 0){ //perform first derivative of input
54  T offset(X.n_rows, X.n_cols);
55  offset.zeros();
56  offset.rows(1, offset.n_rows - 1) = X.rows(0, X.n_rows - 2);

71 template <typename T> T Vespucci::Math::rotate(const T &X, arma::uword shift_by, bool slide_back)
72 {
73  T start;
74  T end;
75 
76  if (shift_by >= X.n_rows){
77  shift_by = shift_by - X.n_rows;
78  }
79 
80  if (shift_by == 0){
81  return X;
82  }
83  else if (shift_by == 1){
84  if (slide_back){
85  start = X.rows(1, X.n_rows - 1);
86  end = X.row(0);
87  }
88  else{
89  start = X.row(X.n_rows - 1);
90  end = X.rows(0, X.n_rows - 2);
91  }
92  }
93  else{
94  if (slide_back){
95  start = X.rows(shift_by, X.n_rows - 1);
96  end = X.rows(0, shift_by - 1);
97  }
98  else{
99  start = X.rows(X.n_rows - shift_by, X.n_rows - 1);
100  end = X.rows(0, X.n_rows - shift_by - 1);
101  }
102  }
103 
104  return join_vert(start, end);
105 }

108 template <typename T> T Vespucci::Math::conv_fft(const T &A, const T &B, std::string type)
109 {
110 
111  if (A.n_rows < B.n_rows){
112  throw std::invalid_argument("B must have the same number of rows or fewer rows than A");
113  }
114 
115  if (type == "filter"){
116  arma::uword next_p2 = std::ceil(std::log(A.n_elem) / std::log(2));
117  arma::uword n = std::pow(2, next_p2);
118  arma::cx_mat A_hat = fft(A, n);
119  arma::cx_mat B_hat = fft(B, n);
120  arma::cx_mat w = ifft(A_hat % B_hat, n);
121  w = w.rows(0, A.n_rows - 1);
122  return real(w);
123  }
124  else if (type == "circular"){
125 
126  T A_padded = join_vert(A, arma::zeros(B.n_rows- 1));
127  T B_padded = join_vert(B, arma::zeros(A.n_rows - 1));
128 
129  arma::uword next_p2 = std::ceil(std::log(A.n_elem) / std::log(2));
130  arma::uword n = std::pow(2, next_p2);
131 
132  arma::cx_mat A_hat = fft(A_padded, n);
133  arma::cx_mat B_hat = fft(B_padded, n);
134  arma::cx_mat w = ifft(A_hat % B_hat, n);
135  //w = w.rows(0, A.n_rows - 1);
136  return real(w);
137  }
138  else
139  throw(std::invalid_argument("argument 'type' invalid"));
140 }
