110 void UnivariateDialog::on_buttonBox_accepted()
111 {
112  if (min_box_->text().isEmpty() || max_box_->text().isEmpty()){
113  QMessageBox::warning(this, "Invalid Input!", "You must enter numbers for left and right bounds.");
114  return;
115  }
116  double entered_min = min_box_->text().toDouble();
117  double entered_max = max_box_->text().toDouble();
118 
119  QString name = name_box_->text();
120  QString value_method = value_method_selector_->currentText();
121  cout << "Method selection" << endl;
122  UnivariateMethod::Method method;
123  if (value_method == "Area")
124  method = UnivariateMethod::Area;
125  else if (value_method == "Bandwidth")
126  method = UnivariateMethod::FWHM;
127  else if (value_method == "Correlation")
128  method = UnivariateMethod::Correlation;
129  else
130  method = UnivariateMethod::Intensity;
131 
132  QString integration_method = integration_method_selector_->currentText();
133  int gradient_index = color_selector_->currentIndex();
134 /*
135  if (entered_min < workspace->GetWavelengthMin(data_index_)){
136  QMessageBox::warning(this, "Invalid Input!", "You have entered a left bound that is smaller than the smallest number on the spectral abscissa");
137  return;
138  }
139 
140  if (entered_max > workspace->GetWavelengthMax(data_index_)){
141  QMessageBox::warning(this, "Invalid Input!", "You have entered a right bound that is larger than the largest number on the spectral abscissa");
142  return;
143  }
144 */
145  if (method != UnivariateMethod::Correlation){
146  try{
147  data_->Univariate(entered_min, entered_max, name, method, integration_method, gradient_index);
148  }
149  catch(exception e){
150  workspace->main_window()->DisplayExceptionWarning(e);
151  }
152  }
153  else{
154  vec control;
155  QFileInfo file_info(file_name_box_->text());
156  workspace->set_directory(file_info.dir().path());
157  try{
158  control.load(file_name_box_->text().toStdString());
159  data_->CorrelationMap(control, name, gradient_index);
160  }
161  catch(exception e){
162  workspace->main_window()->DisplayExceptionWarning(e);
163  }
164  }
165  this->close();
166  data_.clear();
167 }
